1//! Error type for inbound message consumption operations.
2
3use swe_edge_message_broker::BrokerError;
4
5/// Errors returned by [`MessageConsumer`](crate::api::traits::MessageConsumer) operations.
6#[derive(Debug, thiserror::Error)]
7pub enum ConsumerError {
8    /// Failed to subscribe to the given topic.
9    #[error("subscribe failed on topic '{topic}': {reason}")]
10    Subscribe {
11        /// The topic that the subscribe was attempted on.
12        topic: String,
13        /// The reason for the failure.
14        reason: String,
15    },
16    /// Consumer is not connected or the broker is unreachable.
17    #[error("consumer unavailable: {0}")]
18    Unavailable(String),
19    /// Connection to the broker failed.
20    #[error("connection failed: {0}")]
21    Connection(String),
22    /// The message stream lagged and messages were dropped.
23    #[error("stream lagged: {0} messages dropped")]
24    StreamLagged(u64),
25}
26
27impl From<BrokerError> for ConsumerError {
28    fn from(e: BrokerError) -> Self {
29        match e {
30            BrokerError::Subscribe { topic, reason } => ConsumerError::Subscribe { topic, reason },
31            BrokerError::Unavailable(m) => ConsumerError::Unavailable(m),
32            BrokerError::Connection(m) => ConsumerError::Connection(m),
33            BrokerError::StreamLagged { count } => ConsumerError::StreamLagged(count),
34            other => ConsumerError::Unavailable(other.to_string()),
35        }
36    }
37}
