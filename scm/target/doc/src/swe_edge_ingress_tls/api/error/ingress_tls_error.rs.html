1//! Error type for TLS configuration loading.
2
3/// Errors that can occur while loading or building a server TLS configuration.
4///
5/// Returned by [`TlsSvc::build_tls_acceptor`](crate::TlsSvc::build_tls_acceptor).
6/// The path that failed is always included in the message so operators can
7/// identify which file is broken without reading source.
8///
9/// # Examples
10///
11/// ```rust
12/// use swe_edge_ingress_tls::IngressTlsError;
13///
14/// let err = IngressTlsError::CertParse("no certificates found in PEM".to_string());
15/// assert!(err.to_string().contains("no certificates"));
16///
17/// // Match on the variant to apply different recovery strategies.
18/// match err {
19///     IngressTlsError::CertParse(msg) => eprintln!("cert parse failed: {msg}"),
20///     _ => {}
21/// }
22/// ```
23#[derive(Debug, thiserror::Error)]
24pub enum IngressTlsError {
25    /// Certificate file could not be opened.
26    #[error("failed to load certificate from {0}: {1}")]
27    CertLoad(String, #[source] std::io::Error),
28
29    /// Certificate PEM data could not be parsed, or the file contained no
30    /// certificates.
31    #[error("failed to parse certificate: {0}")]
32    CertParse(String),
33
34    /// Private key file could not be opened.
35    #[error("failed to load private key from {0}: {1}")]
36    KeyLoad(String, #[source] std::io::Error),
37
38    /// Private key PEM data could not be parsed, or the file contained no
39    /// key.
40    #[error("failed to parse private key: {0}")]
41    KeyParse(String),
42
43    /// The TLS server configuration could not be built from the provided
44    /// certificate and key materials.
45    #[error("TLS configuration error: {0}")]
46    Config(String),
47}
