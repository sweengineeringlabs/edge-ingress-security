1//! TLS SAF — factory method implementations on [`TlsSvc`].

3use swe_edge_configbuilder::ConfigLoaderFactory;
4
5use crate::api::error::ingress_tls_error::IngressTlsError;
6use crate::api::types::TlsSvc;
7use crate::api::vo::ingress_tls_config::IngressTlsConfig;
8use crate::spi::acceptor::rustls::RustlsAcceptorBuilder;
9
10impl TlsSvc {
11    /// Return a config builder pre-seeded with this crate's package name and version.
12    ///
13    /// # Examples
14    ///
15    /// ```rust,no_run
16    /// use swe_edge_ingress_tls::TlsSvc;
17    ///
18    /// let loader = TlsSvc::create_config_builder()
19    ///     .with_config_dir("config/")
20    ///     .build_loader()
21    ///     .expect("config dir accessible");
22    /// ```
23    pub fn create_config_builder() -> swe_edge_configbuilder::ConfigBuilderImpl {
24        let builder = ConfigLoaderFactory::create_config_builder();
25        builder
26            .with_name(env!("CARGO_PKG_NAME"))
27            .with_version(env!("CARGO_PKG_VERSION"))
28    }
29
30    /// Construct a [`tokio_rustls::TlsAcceptor`] from `config`.
31    ///
32    /// Reads both PEM files eagerly — failures surface here at startup, not at
33    /// connection time. The acceptor is backed by the `ring` CryptoProvider
34    /// without installing a process-wide default.
35    ///
36    /// # Examples
37    ///
38    /// ```rust,no_run
39    /// use swe_edge_ingress_tls::{IngressTlsConfig, TlsSvc};
40    ///
41    /// let config = IngressTlsConfig::tls("certs/server.crt", "certs/server.key");
42    /// let acceptor = TlsSvc::build_tls_acceptor(&config)
43    ///     .expect("PEM files must exist and be valid");
44    /// // pass acceptor to TonicGrpcServer or AxumHttpServer .with_tls()
45    /// ```
46    pub fn build_tls_acceptor(
47        config: &IngressTlsConfig,
48    ) -> Result<tokio_rustls::TlsAcceptor, IngressTlsError> {
49        let acceptor = RustlsAcceptorBuilder::build(config)?;
50        Ok(acceptor)
51    }
52
53    /// Validate any value implementing the [`Validator`](crate::api::traits::validator::Validator)
54    /// contract, returning a human-readable error describing the first failure.
55    pub fn validate<V: crate::api::traits::validator::Validator>(v: &V) -> Result<(), String> {
56        v.validate()
57    }
