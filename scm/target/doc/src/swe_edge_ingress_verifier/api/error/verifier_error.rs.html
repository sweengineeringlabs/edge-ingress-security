1//! `VerifierError` — error type for all verifier operations.
2
3/// Errors returned by [`TokenVerifier`](crate::api::traits::token_verifier::TokenVerifier)
4/// and [`ApiKeyVerifier`](crate::api::types::api_key_verifier::ApiKeyVerifier).
5///
6/// All variants except [`Config`](VerifierError::Config) indicate a caller
7/// error (bad token, expired token, wrong key) — return HTTP 401 to the client.
8/// `Config` indicates a server misconfiguration — return HTTP 500 and page on-call.
9///
10/// # Examples
11///
12/// ```rust
13/// use swe_edge_ingress_verifier::VerifierError;
14///
15/// let err = VerifierError::Expired;
16/// assert!(err.to_string().contains("expired"));
17///
18/// // Map to HTTP status codes.
19/// let status = match err {
20///     VerifierError::Invalid(_)
21///     | VerifierError::Expired
22///     | VerifierError::NotYetValid
23///     | VerifierError::UnknownApiKey => 401,
24///     VerifierError::ClaimMismatch(_) => 403,
25///     VerifierError::Config(_) => 500,
26/// };
27/// assert_eq!(status, 401);
28/// ```
29#[derive(Debug, thiserror::Error)]
30pub enum VerifierError {
31    /// Token signature or structure is invalid.
32    #[error("invalid token: {0}")]
33    Invalid(String),
34    /// Token has expired.
35    #[error("token expired")]
36    Expired,
37    /// `nbf` claim is in the future.
38    #[error("token not yet valid")]
39    NotYetValid,
40    /// Required claim (`iss`, `aud`, `sub`) does not match expected value.
41    #[error("claim mismatch: {0}")]
42    ClaimMismatch(String),
43    /// API key was not recognised.
44    #[error("unknown api key")]
45    UnknownApiKey,
46    /// Verifier configuration is invalid (wrong key format, unsupported algorithm, etc.).
47    #[error("verifier config error: {0}")]
48    Config(String),
49}
50
51impl From<VerifierError> for edge_domain::HandlerError {
52    fn from(e: VerifierError) -> Self {
53        match e {
54            VerifierError::Invalid(_)
55            | VerifierError::Expired
56            | VerifierError::NotYetValid
57            | VerifierError::UnknownApiKey => {
58                edge_domain::HandlerError::Unauthorized(e.to_string())
59            }
60            VerifierError::ClaimMismatch(_) => {
61                edge_domain::HandlerError::PermissionDenied(e.to_string())
62            }
63            VerifierError::Config(_) => edge_domain::HandlerError::ExecutionFailed(e.to_string()),
64        }
65    }
66}
