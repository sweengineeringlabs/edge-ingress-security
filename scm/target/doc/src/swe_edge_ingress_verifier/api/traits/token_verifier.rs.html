2
3use crate::api::error::VerifierError;
4use crate::api::vo::Claims;
5
6/// Verifies an inbound bearer token string and returns its claims.
7///
8/// Implementations decide the algorithm (HS256, RS256, ES256), key
9/// material, and which claims to enforce. The trait is object-safe so
10/// callers can store `Arc<dyn TokenVerifier>` and swap implementations
11/// (e.g. swap a noop stub in tests for a real JWT verifier in production).
12///
13/// The built-in implementation is [`JwtVerifier`](crate::JwtVerifier).
14///
15/// # Examples
16///
17/// ```rust,no_run
18/// use std::sync::Arc;
19/// use swe_edge_ingress_verifier::{Claims, TokenVerifier, VerifierError};
20///
21/// /// Stub verifier that accepts any token in tests.
22/// struct AlwaysAllow;
23/// impl TokenVerifier for AlwaysAllow {
24///     fn verify(&self, _token: &str) -> Result<Claims, VerifierError> {
25///         Ok(Claims::builder().sub("test-user").build())
26///     }
27/// }
28///
29/// let verifier: Arc<dyn TokenVerifier> = Arc::new(AlwaysAllow);
30/// let claims = verifier.verify("any.token.here").unwrap();
31/// assert_eq!(claims.sub.as_deref(), Some("test-user"));
32/// ```
33pub trait TokenVerifier: Send + Sync {
34    /// Verify `token` and return the extracted [`Claims`] on success.
35    ///
36    /// # Examples
37    ///
38    /// ```rust,no_run
39    /// use swe_edge_ingress_verifier::{JwtConfig, JwtKey, JwtVerifier, TokenVerifier};
40    ///
41    /// # let cfg = JwtConfig { key: JwtKey::Hs256 { secret: vec![] }, required_issuer: None, required_audience: None, leeway_seconds: 0 };
42    /// let verifier = JwtVerifier::from_config(&cfg).unwrap();
43    /// // verifier.verify("eyJ...") returns Ok(Claims) or Err(VerifierError::Expired) etc.
44    /// ```
45    fn verify(&self, token: &str) -> Result<Claims, VerifierError>;
46}
