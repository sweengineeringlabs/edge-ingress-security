3use serde::{Deserialize, Serialize};

5/// Key material used to verify JWT signatures.
6///
7/// Passed as the `key` field of [`JwtConfig`](crate::JwtConfig).
8/// Choose the variant that matches the algorithm your identity provider uses:
9///
10/// | Variant | Algorithm | Key format |
11/// |---------|-----------|------------|
12/// | `Hs256` | HS256 | shared secret bytes |
13/// | `Rs256` | RS256 | PEM RSA public key |
14/// | `Es256` | ES256 | PEM EC public key |
15///
16/// # Examples
17///
18/// ```rust
19/// use swe_edge_ingress_verifier::JwtKey;
20///
21/// // HS256 — symmetric shared secret.
22/// let key = JwtKey::Hs256 { secret: b"super-secret".to_vec() };
23///
24/// // RS256 — asymmetric RSA public key.
25/// let key = JwtKey::Rs256 { public_pem: "-----BEGIN PUBLIC KEY-----\n...\n-----END PUBLIC KEY-----".to_string() };
26///
27/// // ES256 — asymmetric EC public key.
28/// let key = JwtKey::Es256 { public_pem: "-----BEGIN PUBLIC KEY-----\n...\n-----END PUBLIC KEY-----".to_string() };
29/// ```
30#[derive(Debug, Clone, Serialize, Deserialize)]
31#[serde(tag = "algorithm", rename_all = "SCREAMING_SNAKE_CASE")]
32pub enum JwtKey {
33    /// HS256 — shared symmetric secret bytes.
34    Hs256 {
35        /// Raw HMAC secret bytes.
36        secret: Vec<u8>,
37    },
38    /// RS256 — PEM-encoded RSA public key.
39    Rs256 {
40        /// PEM-encoded RSA public key.
41        public_pem: String,
42    },
43    /// ES256 — PEM-encoded EC public key.
44    Es256 {
45        /// PEM-encoded EC public key.
46        public_pem: String,
47    },
48}
